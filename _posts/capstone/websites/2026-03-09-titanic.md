			<p id="improvement-summary">Suggested changes will appear here.</p>
			<ul id="improvement-list">
				<li>No improvement plan yet.</li>
			</ul>
		</section>
	</section>
</div>

<script>
	const LOCAL_PAGE_PORTS = new Set(['4000', '4500', '4599', '4600']);
	const LOCAL_API_PORT = '8587';

	function resolveApiBase() {
		const { protocol, hostname, port, origin } = window.location;
		const isLocalHost = hostname === 'localhost' || hostname === '127.0.0.1';
		const isFlaskServedPage = isLocalHost && (port === '8367' || port === LOCAL_API_PORT);

		if (isFlaskServedPage) {
			return `${origin}/api/titanic`;
		}

		if (isLocalHost && LOCAL_PAGE_PORTS.has(port)) {
			return `http://${hostname}:${LOCAL_API_PORT}/api/titanic`;
		}

		if (protocol === 'https:') {
			return 'https://flask.opencodingsociety.com/api/titanic';
		}

		if (isLocalHost) {
			return `http://${hostname}:${LOCAL_API_PORT}/api/titanic`;
		}

		return `${origin}/api/titanic`;
	}

	const apiBase = resolveApiBase();

	const form = document.getElementById('titanic-form');
	const statusEl = document.getElementById('status');
	const accuracyEl = document.getElementById('accuracy');
	const bestEmbarkedEl = document.getElementById('best-embarked');

	const surviveValueEl = document.getElementById('survive-value');
	const dieValueEl = document.getElementById('die-value');
	const surviveBarEl = document.getElementById('survive-bar');
	const dieBarEl = document.getElementById('die-bar');

	const topFactorsEl = document.getElementById('top-factors');
	const improvementSummaryEl = document.getElementById('improvement-summary');
	const improvementListEl = document.getElementById('improvement-list');
	const titanicSceneEl = document.getElementById('titanic-scene');
	const titanicSceneCaptionEl = document.getElementById('titanic-scene-caption');

	function formatPercent(value) {
		return `${(value * 100).toFixed(1)}%`;
	}

	function setStatus(message, isError = false) {
		statusEl.textContent = message;
		statusEl.className = isError ? 'titanic-status error' : 'titanic-status';
	}

	function updateBoatState(surviveProbability = null) {
		let state = 'cruising';
		let caption = 'The ship is cruising while the forecast loads.';

		if (typeof surviveProbability === 'number') {
			if (surviveProbability < 0.3) {
				state = 'sinking';
				caption = 'Danger zone: the ship slams into an iceberg, then sinks with odds below 30%.';
			} else if (surviveProbability < 0.7) {
				state = 'shaky';
				caption = 'Unsteady waters: the ship rocks hard with mid-range survival odds.';
			} else {
				state = 'thriving';
				caption = 'Clear sailing: the ship is thriving with strong survival odds.';
			}
		}

		titanicSceneEl.dataset.state = 'reset';
		void titanicSceneEl.offsetWidth;
		titanicSceneEl.dataset.state = state;
		titanicSceneCaptionEl.textContent = caption;
	}

	function buildDiagnosticMessage(error, url) {
		const pageOrigin = window.location.origin;
		const mixedContent = window.location.protocol === 'https:' && url.startsWith('http://');
		const baseMessage = error instanceof Error ? error.message : String(error);

		if (mixedContent) {
			return `${baseMessage} Page origin: ${pageOrigin}. API URL: ${url}. This page is running over HTTPS, so the API must also use HTTPS.`;
		}

		return `${baseMessage} Page origin: ${pageOrigin}. API URL: ${url}. Check the resolved backend URL and confirm this page origin is allowed by Flask CORS.`;
	}

	async function parseJson(response) {
		const text = await response.text();
		return text ? JSON.parse(text) : {};
	}

	async function requestJson(url, options = {}) {
		try {
			const response = await fetch(url, {
				mode: 'cors',
				...options
			});
			const data = await parseJson(response);

			if (!response.ok) {
				throw new Error(data.message || `Request failed with status ${response.status}.`);
			}

			return data;
		} catch (error) {
			throw new Error(buildDiagnosticMessage(error, url));
		}
	}

	async function loadMetadata() {
		try {
			const data = await requestJson(`${apiBase}/meta`);

			accuracyEl.textContent = formatPercent(data.accuracy || 0);
			bestEmbarkedEl.textContent = data.best_embarked || 'Unknown';
			updateBoatState(0.72);
			setStatus(`Model ready. API base: ${apiBase}`);
		} catch (error) {
			updateBoatState(null);
			setStatus(buildDiagnosticMessage(error, `${apiBase}/meta`), true);
		}
	}

	function renderFactors(factors) {
		topFactorsEl.innerHTML = '';

		if (!factors || !factors.length) {
			topFactorsEl.innerHTML = '<li>No factor details returned.</li>';
			return;
		}

		factors.forEach((factor) => {
			const li = document.createElement('li');
			li.textContent =
				`${factor.label} ${factor.direction === 'helps' ? 'helped' : 'hurt'} ` +
				`the prediction most with impact ${factor.impact.toFixed(2)}.`;
			topFactorsEl.appendChild(li);
		});
	}

	function renderImprovement(improvement) {
		improvementListEl.innerHTML = '';

		if (!improvement || !improvement.changes || !improvement.changes.length) {
			improvementSummaryEl.textContent =
				'The model did not find a stronger scenario within its limited adjustment rules.';
			improvementListEl.innerHTML =
				'<li>This passenger profile is already near the strongest conditions the demo checks.</li>';
			return;
		}

		improvementSummaryEl.textContent =
			`A stronger scenario raises survival odds by ${formatPercent(improvement.delta)} to ${formatPercent(improvement.survive)}.`;

		improvement.changes.forEach((entry) => {
			const li = document.createElement('li');
			const changeText = entry.changes
				.map((change) => `${change.field}: ${change.from} to ${change.to}`)
				.join(', ');

			li.textContent = `${entry.reason} Changes: ${changeText}.`;
			improvementListEl.appendChild(li);
		});
	}

	form.addEventListener('submit', async (event) => {
		event.preventDefault();

		const formData = new FormData(form);
		const payload = {
			pclass: Number(formData.get('pclass')),
			sex: formData.get('sex'),
			age: Number(formData.get('age')),
			sibsp: Number(formData.get('sibsp')),
			parch: Number(formData.get('parch')),
			fare: Number(formData.get('fare')),
			embarked: formData.get('embarked'),
			alone: formData.get('alone') === 'true'
		};

		setStatus('Running prediction...');

		try {
			const data = await requestJson(`${apiBase}/predict`, {
				method: 'POST',
				headers: {
					'Content-Type': 'application/json'
				},
				body: JSON.stringify(payload)
			});

			surviveValueEl.textContent = formatPercent(data.survive);
			dieValueEl.textContent = formatPercent(data.die);
			updateBoatState(data.survive);

			surviveBarEl.style.width = `${data.survive * 100}%`;
			dieBarEl.style.width = `${data.die * 100}%`;

			renderFactors(data.top_factors);
			renderImprovement(data.improvement);

			setStatus(`Prediction complete. Model accuracy is ${formatPercent(data.accuracy || 0)}.`);
		} catch (error) {
			updateBoatState(null);
			setStatus(buildDiagnosticMessage(error, `${apiBase}/predict`), true);
		}
	});

	loadMetadata();
</script>
